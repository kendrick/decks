Lists & Maps

Lists
$states: 'success', 'error',
         'warning', 'info';
Maps
$states-with-colors: (
  success: #00b575,
  error: #d51c31,
  warning: #ffc300,
  info: #1492d9
);
Maps
map-get($states-with-colors, 'success');

Functions

Functions
@function get-state-color($state) {
  @return map-get($states-with-colors,
                  $state);
}

Mixins

Mixins
@mixin state-color($state) {
  color: get-state-color($state);
}
Mixins
.error {
  @include state-color('error');
}

.success {
  @include state-color('success');
}
Mixins (Bonus!)
@each $state in $states {
  .#{$state} {
    @include state-color($state);
  }
}
